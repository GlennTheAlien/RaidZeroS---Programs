task autonomous()

	motor[bottomRight] = 0;
}

/*---------------------------------------------------------------------------*/
/*                                                                           */
/*                              User Control Task                            */
/*                                                                           */
/*  This task is used to control your robot during the user control phase of */
/*  a VEX Competition.                                                       */
/*                                                                           */
/*  You must modify the code to add your own robot specific commands here.   */
/*---------------------------------------------------------------------------*/

task usercontrol()
{
	// User control code here, inside the loop

	while (true)
	{
		//Drive Control//
		motor[topLeft]=vexRT[Ch3];
		motor[bottomLeft]=vexRT[Ch3];
		motor[topRight]=vexRT[Ch2];
		motor[bottomRight]=vexRT[Ch2];

		//Arm Control//
		if(vexRT[Btn6U]==1)
		{
			motor[arm1]=100;
			motor[arm2]=100;
		}
		else if(vexRT[Btn6D]==1)
		{
			motor[arm1]=-100;
			motor[arm2]=-100;
		}
		else
		{
			motor[arm1]=0;
			motor[arm2]=0;
		}

		//Claw Up-Down
		if(vexRT[Btn5U]==1)
		{
			motor[upClaw]=127;
		}
		else if(vexRT[Btn5D]==1)
		{
			motor[upClaw]=-127;
		}
		else
		{
			motor[upClaw]=0;
		}

		//Claw Open-Close//
		if(vexRT[Btn7D]==1)
		{
			motor[openClaw]=127;
		}
		else if(vexRT[Btn7U]==1)
		{
			motor[openClaw]=-127;
		}
		else
		{
			motor[openClaw]=0;
		}

		//Shoot Mechanism//
		if(vexRT[Btn8D]==1)
		{
			motor[shoot1]=100;
			motor[shoot2]=100;
		}
		else
		{
			motor[shoot1]=0;
			motor[shoot2]=0;
		}
	}
}
